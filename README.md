import turtle
screen=turtle.Screen()
screen.bgcolor("black")
shape=turtle.Turtle()
shape.color("blue","lightblue")
shape.begin_fill()
shape.circle(100)
shape.end_fill()
shape.penup()
shape.goto(0,50)
shape.pendown()
shape.color("white","black")
shape.begin_fill()
shape.circle(50)
shape.end_fill()
shape.penup()
shape.goto(0,75)
shape.pendown()
shape.color("white","white")
shape.begin_fill()
shape.circle(25)
turtle.done()
